1. QWidget vs QMainWindow

Dalam PySide6, ada dua pilihan utama untuk membuat window aplikasi: QWidget dan QMainWindow. Pemilihan yang tepat sangat penting untuk membangun aplikasi yang sesuai kebutuhan.

QWidget
  • Base class untuk semua UI objects
  • Ringan dan sederhana
  • Cocok untuk dialog atau window kecil
  • Tidak ada built-in menu/toolbar
  • Layout harus diatur manual
  • Contoh: form input, dialog kustom
QMainWindow
  • Framework untuk main window
  • Built-in MenuBar, ToolBar, StatusBar
  • Support DockWidget
  • Central widget area
  • Cocok untuk aplikasi utama
  • Contoh: text editor, IDE, browser

Kapan Menggunakan Masing-masing?

Kriteria Gunakan QWidget Gunakan QMainWindow
Aplikasi sederhana ✓ Ya Tidak perlu
Butuh menu bar Bisa, tapi manual ✓ Built-in
Butuh toolbar Bisa, tapi manual ✓ Built-in
Butuh status bar Bisa, tapi manual ✓ Built-in
Aplikasi profesional Tidak ideal ✓ Sangat cocok
Dialog/popup ✓ Ideal Tidak cocok

Implementasi Lengkap Struktur QMainWindow

mainwindow_structure.py Python 
import sys
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QTextEdit, QLabel, QDockWidget, QListWidget
)
from PySide6.QtCore import Qt

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Struktur QMainWindow")
        self.setGeometry(100, 100, 800, 600)
        
        self.setup_central_widget()
        self.setup_menu_bar()
        self.setup_toolbars()
        self.setup_status_bar()
        self.setup_dock_widgets()
    
    def setup_central_widget(self):
        """Membuat central widget - area konten utama"""
        self.text_edit = QTextEdit()
        self.text_edit.setPlaceholderText("Ketik teks di sini...")
        self.setCentralWidget(self.text_edit)
    
    def setup_menu_bar(self):
        """Membuat menu bar"""
        menu_bar = self.menuBar()
        
        # Menu File
        file_menu = menu_bar.addMenu("&File")
        file_menu.addAction("New")
        file_menu.addAction("Open")
        file_menu.addAction("Save")
        file_menu.addSeparator()
        file_menu.addAction("Exit")
        
        # Menu Edit
        edit_menu = menu_bar.addMenu("&Edit")
        edit_menu.addAction("Undo")
        edit_menu.addAction("Redo")
        edit_menu.addSeparator()
        edit_menu.addAction("Cut")
        edit_menu.addAction("Copy")
        edit_menu.addAction("Paste")
        
        # Menu View
        view_menu = menu_bar.addMenu("&View")
        view_menu.addAction("Zoom In")
        view_menu.addAction("Zoom Out")
        
        # Menu Help
        help_menu = menu_bar.addMenu("&Help")
        help_menu.addAction("About")
    
    def setup_toolbars(self):
        """Membuat toolbar"""
        # Toolbar File
        file_toolbar = self.addToolBar("File")
        file_toolbar.addAction("New")
        file_toolbar.addAction("Open")
        file_toolbar.addAction("Save")
        
        # Toolbar Edit
        edit_toolbar = self.addToolBar("Edit")
        edit_toolbar.addAction("Cut")
        edit_toolbar.addAction("Copy")
        edit_toolbar.addAction("Paste")
    
    def setup_status_bar(self):
        """Membuat status bar"""
        self.statusBar().showMessage("Ready")
        
        # Menambahkan widget permanen ke status bar
        self.status_label = QLabel("Ln 1, Col 1")
        self.statusBar().addPermanentWidget(self.status_label)
    
    def setup_dock_widgets(self):
        """Membuat dock widgets"""
        # Dock kiri - File Explorer
        file_dock = QDockWidget("File Explorer", self)
        file_dock.setAllowedAreas(Qt.LeftDockWidgetArea | Qt.RightDockWidgetArea)
        
        file_list = QListWidget()
        file_list.addItems(["document.txt", "image.png", "data.csv"])
        file_dock.setWidget(file_list)
        
        self.addDockWidget(Qt.LeftDockWidgetArea, file_dock)
        
        # Dock kanan - Properties
        prop_dock = QDockWidget("Properties", self)
        prop_dock.setAllowedAreas(Qt.LeftDockWidgetArea | Qt.RightDockWidgetArea)
        
        prop_widget = QWidget()
        prop_layout = QVBoxLayout(prop_widget)
        prop_layout.addWidget(QLabel("File: document.txt"))
        prop_layout.addWidget(QLabel("Size: 1.2 KB"))
        prop_layout.addWidget(QLabel("Modified: 2024-01-15"))
        prop_layout.addStretch()
        prop_dock.setWidget(prop_widget)
        
        self.addDockWidget(Qt.RightDockWidgetArea, prop_dock)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = MainWindow()
    window.show()
    sys.exit(app.exec())
Tentang Ampersand (&) pada Menu

Karakter & sebelum huruf pada nama menu (seperti "&File") membuat huruf tersebut menjadi mnemonic/accelerator. User bisa mengakses menu dengan Alt+F untuk File, Alt+E untuk Edit, dll.

3. QAction: Menu + Toolbar + Shortcut

QAction adalah abstraksi untuk aksi yang bisa dipicu dari berbagai tempat: menu, toolbar, atau keyboard shortcut. Ini memungkinkan satu definisi aksi untuk multiple UI elements.

Properti Penting QAction

Properti/Method Deskripsi Contoh
setText() Teks yang ditampilkan "Save"
setShortcut() Keyboard shortcut QKeySequence.Save
setIcon() Icon untuk action QIcon("save.png")
setStatusTip() Teks di status bar saat hover "Save the document"

Implementasi QAction

qaction_demo.py Python 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QTextEdit
from PySide6.QtGui import QAction, QKeySequence

class TextEditor(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("QAction Demo")
        self.setGeometry(100, 100, 700, 500)

        self.text_edit = QTextEdit()
        self.setCentralWidget(self.text_edit)

        self.create_actions()
        self.create_menus()
        self.create_toolbars()
        self.statusBar().showMessage("Ready")

    def create_actions(self):
        # New Action
        self.new_action = QAction("&New", self)
        self.new_action.setShortcut(QKeySequence.New)
        self.new_action.setStatusTip("Buat dokumen baru")
        self.new_action.triggered.connect(self.new_document)

        # Save Action
        self.save_action = QAction("&Save", self)
        self.save_action.setShortcut(QKeySequence.Save)
        self.save_action.setStatusTip("Simpan dokumen")
        self.save_action.triggered.connect(self.save_document)

        # Exit Action
        self.exit_action = QAction("E&xit", self)
        self.exit_action.setShortcut("Ctrl+Q")
        self.exit_action.setStatusTip("Keluar dari aplikasi")
        self.exit_action.triggered.connect(self.close)

        # Word Wrap — contoh checkable action (bisa di-toggle on/off)
        self.wordwrap_action = QAction("&Word Wrap", self)
        self.wordwrap_action.setCheckable(True)
        self.wordwrap_action.setChecked(True)
        self.wordwrap_action.setStatusTip("Toggle word wrap")
        self.wordwrap_action.triggered.connect(self.toggle_word_wrap)

    def create_menus(self):
        # File Menu — action yang sama dipasang di sini...
        file_menu = self.menuBar().addMenu("&File")
        file_menu.addAction(self.new_action)
        file_menu.addAction(self.save_action)
        file_menu.addSeparator()
        file_menu.addAction(self.exit_action)

        # View Menu — contoh checkable action di menu
        view_menu = self.menuBar().addMenu("&View")
        view_menu.addAction(self.wordwrap_action)

    def create_toolbars(self):
        # ...dan action yang SAMA juga dipasang di toolbar — tidak perlu definisi ulang
        toolbar = self.addToolBar("File")
        toolbar.addAction(self.new_action)
        toolbar.addAction(self.save_action)

    # === SLOT METHODS ===

    def new_document(self):
        self.text_edit.clear()
        self.statusBar().showMessage("Dokumen baru dibuat", 3000)

    def save_document(self):
        self.statusBar().showMessage("Dokumen disimpan", 3000)

    def toggle_word_wrap(self, checked):
        if checked:
            self.text_edit.setLineWrapMode(QTextEdit.WidgetWidth)
            self.statusBar().showMessage("Word wrap ON", 2000)
        else:
            self.text_edit.setLineWrapMode(QTextEdit.NoWrap)
            self.statusBar().showMessage("Word wrap OFF", 2000)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = TextEditor()
    window.show()
    sys.exit(app.exec())
Keuntungan Menggunakan QAction
  • Reusability: Satu action bisa dipakai di menu dan toolbar
  • Consistency: Shortcut, icon, dan text selalu konsisten
  • Easy Management: Enable/disable dari satu tempat
  • State Management: Checkable actions untuk toggle state

5. Dialog Bawaan: QMessageBox, QFileDialog, QInputDialog

PySide6 menyediakan berbagai dialog bawaan untuk keperluan umum seperti menampilkan pesan, memilih file, atau meminta input dari user.

QMessageBox

Dialog untuk menampilkan pesan informatif, peringatan, error, atau pertanyaan kepada user.

Tipe Method Kegunaan
Information QMessageBox.information() Pesan informasi umum
Warning QMessageBox.warning() Pesan peringatan
Critical QMessageBox.critical() Pesan error/kritis
Question QMessageBox.question() Pertanyaan Yes/No
About QMessageBox.about() Dialog about aplikasi

QFileDialog

Dialog untuk membuka atau menyimpan file.

Method Return Kegunaan
getOpenFileName() (path, filter) Pilih satu file untuk dibuka
getOpenFileNames() (paths, filter) Pilih multiple files
getSaveFileName() (path, filter) Pilih lokasi save file
getExistingDirectory() path Pilih folder

QInputDialog

Dialog untuk meminta input sederhana dari user.

Method Return Kegunaan
getText() (text, ok) Input teks
getInt() (value, ok) Input integer
getDouble() (value, ok) Input decimal
getItem() (item, ok) Pilih dari list

    def show_info(self):
        QMessageBox.information(
            self,
            "Informasi",
            "Ini adalah pesan informasi.\nOperasi berhasil dilakukan!"
        )
        self.result_label.setText("Information dialog ditampilkan")
    
    def show_warning(self):
        QMessageBox.warning(
            self,
            "Peringatan",
            "Ini adalah pesan peringatan!\nHarap perhatikan."
        )
        self.result_label.setText("Warning dialog ditampilkan")
    
    def show_critical(self):
        QMessageBox.critical(
            self,
            "Error",
            "Terjadi kesalahan kritis!\nAplikasi tidak dapat melanjutkan."
        )
        self.result_label.setText("Critical dialog ditampilkan")
    
    def show_question(self):
        reply = QMessageBox.question(

            self.result_label.setText("Anda memilih: CANCEL")

Ketika dialog bawaan tidak mencukupi kebutuhan, kita bisa membuat custom dialog dengan mewarisi QDialog.

Konsep Penting QDialog

Method/Property Deskripsi
exec() Menampilkan dialog secara modal (blocking)
show() Menampilkan dialog secara non-modal
accept() Menutup dialog dengan status Accepted
reject() Menutup dialog dengan status Rejected

Modal: User harus menutup dialog sebelum bisa berinteraksi dengan window utama (exec())

    def new_file(self):
        """Membuat file baru"""
        # Cek dulu apakah ada perubahan yang belum disimpan
        if self.maybe_save():
            self.text_edit.clear()
            self.current_file = None
            self.is_modified = False
            self.update_title()
            self.statusBar().showMessage("New file created", 3000)
    
    def open_file(self):
        """Membuka file yang sudah ada"""
        # Cek dulu apakah ada perubahan yang belum disimpan
        if self.maybe_save():
            file_path, _ = QFileDialog.getOpenFileName(
                self,
                "Open File",
                "",
                "Text Files (*.txt);;Python Files (*.py);;All Files (*)"
            )
            
            if file_path:
                self.load_file(file_path)
    
    def load_file(self, file_path):
        """Load file ke text editor"""
        try:
            with open(file_path, 'r', encoding='utf-8') as file:
                content = file.read()
            
            self.text_edit.setPlainText(content)
            self.current_file = file_path
            self.is_modified = False
            self.update_title()
            self.statusBar().showMessage(f"Loaded: {file_path}", 3000)
        
        except Exception as e:
            QMessageBox.critical(
                self,
                "Error",
                f"Cannot open file:\n{str(e)}"
            )
    
    def save_file(self):
        """Menyimpan file"""
        if self.current_file:
            # Jika sudah ada file, simpan langsung
            self.write_file(self.current_file)
        else:
            # Jika belum ada file, tampilkan Save As
            self.save_file_as()
    
    def save_file_as(self):
        """Menyimpan file dengan nama baru"""
        file_path, _ = QFileDialog.getSaveFileName(
            self,
            "Save File",
            self.current_file or "untitled.txt",
            "Text Files (*.txt);;Python Files (*.py);;All Files (*)"
        )
        
        if file_path:
            self.write_file(file_path)
    
    def write_file(self, file_path):
        """Menulis konten ke file"""
        try:
            content = self.text_edit.toPlainText()
            
            with open(file_path, 'w', encoding='utf-8') as file:
                file.write(content)
            
            self.current_file = file_path
            self.is_modified = False
            self.update_title()
            self.statusBar().showMessage(f"Saved: {file_path}", 3000)
        
        except Exception as e:
            QMessageBox.critical(
                self,
                "Error",
                f"Cannot save file:\n{str(e)}"
            )
    
    # ===== HELPER METHODS =====
    
    def maybe_save(self):
        """
        Cek apakah perlu menyimpan perubahan.
        Return True jika boleh melanjutkan (sudah disimpan/discard).
        Return False jika user memilih Cancel.
        """
        if not self.is_modified:
            return True
        
        reply = QMessageBox.question(
            self,
            "Unsaved Changes",
            "Document has been modified.\nDo you want to save changes?",
            QMessageBox.Save | QMessageBox.Discard | QMessageBox.Cancel,
            QMessageBox.Save
        )
        
        if reply == QMessageBox.Save:
            self.save_file()
            # Return True hanya jika berhasil disimpan
            return not self.is_modified
        elif reply == QMessageBox.Discard:
            return True
        else:  # Cancel
            return False
    
    def on_text_changed(self):
        """Handler saat text berubah"""
        self.is_modified = True
        self.update_title()
    
    def update_title(self):
        """Update window title dengan nama file dan status modified"""
        if self.current_file:
            filename = os.path.basename(self.current_file)
        else:
            filename = "Untitled"
        
        # Tambahkan asterisk (*) jika ada perubahan
        modified_marker = " *" if self.is_modified else ""
        self.setWindowTitle(f"{filename}{modified_marker} - Text Editor")
    
    def closeEvent(self, event):
        """Override closeEvent untuk konfirmasi sebelum tutup"""
        if self.maybe_save():
            event.accept()
        else:
            event.ignore()

if __name__ == "__main__":
    app = QApplication(sys.argv)
    editor = TextEditor()
    editor.show()

Penjelasan Fitur-fitur Penting

1

State Tracking (current_file & is_modified)

current_file menyimpan path file yang sedang aktif. is_modified adalah flag untuk mendeteksi apakah ada perubahan yang belum disimpan.

2

on_text_changed Handler

Setiap kali user mengetik, signal textChanged dipancarkan. Kita gunakan ini untuk set is_modified = True dan update judul window.

3

maybe_save() Method

Method ini dipanggil sebelum New, Open, atau Close. Fungsinya untuk menanyakan user apakah ingin menyimpan perubahan. Return True jika boleh melanjutkan, False jika dibatalkan.

4

update_title() Method

Update judul window dengan format: filename * - Text Editor. Asterisk (*) muncul jika ada perubahan yang belum disimpan.

5

closeEvent Override

Override method closeEvent untuk mencegah window ditutup jika ada perubahan yang belum disimpan. User akan ditanya dulu.

Ringkasan Fitur yang Diimplementasikan
  • New File: Membuat dokumen baru dengan pengecekan unsaved changes
  • Open File: Membuka file dengan QFileDialog dan filter tipe file
  • Save File: Menyimpan ke file yang sama jika sudah ada, atau Save As jika baru
  • Save As: Selalu menampilkan dialog untuk memilih nama/lokasi baru
  • Modified Tracking: Mendeteksi perubahan dan menandai dengan asterisk (*)
  • Close Protection: Konfirmasi sebelum menutup jika ada perubahan
  • Error Handling: Menampilkan pesan error jika gagal baca/tulis file

  • Gunakan with statement untuk memastikan file selalu ditutup
  • Tangani exception dengan try-except dan tampilkan pesan error yang jelas
  • Implementasikan "dirty flag" (is_modified) untuk tracking perubahan

class ThemeApp(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Light/Dark Theme Demo")
        self.setGeometry(100, 100, 600, 450)
        self.is_dark = False  # State tema saat ini

        self.setup_ui()
        self.setup_menu()
        self.apply_theme()  # Terapkan tema awal

    def setup_ui(self):
        central = QWidget()
        self.setCentralWidget(central)
        layout = QVBoxLayout(central)

        self.theme_label = QLabel("Tema Aktif: Light Mode ☀️")
        self.theme_label.setStyleSheet("font-size: 16px; font-weight: bold; padding: 10px;")

        self.text_edit = QTextEdit()
        self.text_edit.setPlaceholderText("Ketik sesuatu di sini...")

        btn_layout = QHBoxLayout()
        self.toggle_btn = QPushButton("🌙 Switch to Dark Mode")
        self.toggle_btn.clicked.connect(self.toggle_theme)
        btn_layout.addWidget(self.toggle_btn)

        layout.addWidget(self.theme_label)
        layout.addWidget(self.text_edit)
        layout.addLayout(btn_layout)

        self.statusBar().showMessage("Light Mode aktif")

    def setup_menu(self):
        view_menu = self.menuBar().addMenu("&View")
        toggle_action = QAction("Toggle Dark Mode", self)
        toggle_action.setShortcut("Ctrl+Shift+T")
        toggle_action.triggered.connect(self.toggle_theme)
        view_menu.addAction(toggle_action)

    def toggle_theme(self):
        """Toggle antara light dan dark theme"""
        self.is_dark = not self.is_dark
        self.apply_theme()

    def apply_theme(self):
        """Terapkan tema sesuai state is_dark"""
        if self.is_dark:
            QApplication.instance().setStyleSheet(DARK_THEME)
            self.theme_label.setText("Tema Aktif: Dark Mode 🌙")
            self.toggle_btn.setText("☀️ Switch to Light Mode")
            self.statusBar().showMessage("Dark Mode aktif")
        else:
            QApplication.instance().setStyleSheet(LIGHT_THEME)
            self.theme_label.setText("Tema Aktif: Light Mode ☀️")
            self.toggle_btn.setText("🌙 Switch to Dark Mode")
            self.statusBar().showMessage("Light Mode aktif")

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = ThemeApp()
    window.show()
    sys.exit(app.exec())
Tips Implementasi Tema
  • Simpan tema di konstanta atau file .qss terpisah agar mudah dikelola

  • Shortcut keyboard (Ctrl+Shift+T) membuat toggle lebih mudah diakses
