class MainWindow(QMainWindow):

🐛 Bug #2 — Variabel Lokal vs Atribut Instance

Jalankan kode di atas. Label mungkin muncul sebentar lalu hilang, atau tidak muncul sama sekali tergantung versi Qt. Tidak ada error message — ini yang bikin mahasiswa bingung.

Tanya ke kelas: "Kode tidak error, tapi label tidak muncul. Kira-kira kenapa?"

Jawaban: variabel label tanpa self hanya hidup selama __init__ berjalan. Begitu method selesai, Python garbage collector bisa menghapusnya karena tidak ada referensi yang menyimpannya. Solusinya: pakai self.label agar objek tetap hidup selama window hidup.

Catatan: Untuk widget yang sudah di-addWidget() ke layout, Qt biasanya menjaga referensinya — tapi kebiasaan pakai self tetap penting untuk atribut yang perlu diakses dari method lain.

📌 Memahami self di Python & PySide6

self adalah referensi ke objek itu sendiri. Ketika kamu menulis self.label, kamu menyimpan label sebagai atribut instance — artinya objek itu "mengingat" label tersebut selama objek masih hidup.

Tanpa self, variabel hanya hidup di dalam method tempat ia dibuat. Begitu method selesai dieksekusi, variabel itu hilang.

Analogi Sederhana

Bayangkan kelas sebagai formulir. self.nama itu seperti mengisi kolom "Nama" di formulir — tersimpan permanen. Sedangkan variabel lokal seperti catatan di kertas coret-coretan — dibuang setelah selesai dipakai.

Aturan Praktis

import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QWidget, QVBoxLayout, QLabel, QPushButton

class ContohSelf(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Contoh self")
        self.setGeometry(100, 100, 400, 300)

        central = QWidget()
        self.setCentralWidget(central)
        layout = QVBoxLayout(central)

        # ✅ PAKAI self — bisa diakses dari method lain
        self.counter_label = QLabel("Klik: 0")
        self.count = 0

        # ⚠️ TANPA self — hanya untuk widget statis yang tidak perlu diubah
        judul = QLabel("Contoh Counter")
        judul.setStyleSheet("font-weight: bold; font-size: 16px;")

        btn = QPushButton("Tambah")
        btn.clicked.connect(self.tambah)  # btn tidak perlu self karena tidak diakses lagi

        layout.addWidget(judul)
        layout.addWidget(self.counter_label)
        layout.addWidget(btn)

    def tambah(self):
        # Method ini butuh self.count dan self.counter_label
        # Kalau tidak pakai self di __init__, baris ini akan error!
        self.count += 1
        self.counter_label.setText(f"Klik: {self.count}")

app = QApplication(sys.argv)
w = ContohSelf()
w.show()
sys.exit(app.exec())

3. QAction ⏱ 20 menit

Cara Mengajar: Tunjukkan masalah dulu — buat action "Save" di menu dan toolbar secara terpisah (dua kali). Lalu tanya: "Kalau ada 10 action, kita harus tulis 20 kali?" Dari sini mahasiswa akan paham kenapa QAction ada.

🔑 Konsep Kunci yang Harus Dipahami

QAction = satu definisi, bisa dipasang di menu DAN toolbar DAN shortcut sekaligus. Ubah satu, semua berubah.

✅ Kode ini bisa di-copy langsung

qaction_demo.py — live coding bertahap Live Coding 
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QTextEdit, QMessageBox
from PySide6.QtGui import QAction, QKeySequence

class TextEditor(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("QAction Demo")
        self.setGeometry(100, 100, 700, 500)
        self.text_edit = QTextEdit()
        self.setCentralWidget(self.text_edit)

        self.create_actions()
        self.create_menus()
        self.create_toolbars()
        self.statusBar().showMessage("Ready")

    def create_actions(self):
        # New
        self.new_action = QAction("&New", self)
        self.new_action.setShortcut(QKeySequence.New)
        self.new_action.setStatusTip("Buat dokumen baru")
        self.new_action.triggered.connect(self.new_doc)

        # Save
        self.save_action = QAction("&Save", self)
        self.save_action.setShortcut(QKeySequence.Save)
        self.save_action.setStatusTip("Simpan dokumen")
        self.save_action.triggered.connect(self.save_doc)

        # Word Wrap — checkable action
        self.wrap_action = QAction("&Word Wrap", self)
        self.wrap_action.setCheckable(True)
        self.wrap_action.setChecked(True)
        self.wrap_action.triggered.connect(self.toggle_wrap)

    def create_menus(self):
        file_menu = self.menuBar().addMenu("&File")
        file_menu.addAction(self.new_action)   # pakai action yang sama
        file_menu.addAction(self.save_action)

        view_menu = self.menuBar().addMenu("&View")
        view_menu.addAction(self.wrap_action)

    def create_toolbars(self):
        toolbar = self.addToolBar("File")
        toolbar.addAction(self.new_action)     # action yang sama lagi
        toolbar.addAction(self.save_action)

    def new_doc(self):
        self.text_edit.clear()
        self.statusBar().showMessage("Dokumen baru dibuat", 3000)

    def save_doc(self):
        self.statusBar().showMessage("Dokumen disimpan", 3000)

    def toggle_wrap(self, checked):
        if checked:
            self.text_edit.setLineWrapMode(QTextEdit.WidgetWidth)
        else:
            self.text_edit.setLineWrapMode(QTextEdit.NoWrap)

app = QApplication(sys.argv)
w = TextEditor()
w.show()
sys.exit(app.exec())

5. Dialog Bawaan ⏱ 20 menit

Cara Mengajar: Demo satu per satu, jalankan langsung. Fokus pada QMessageBox.question() karena punya return value yang perlu dicek — ini sering membingungkan mahasiswa.

✅ Kode ini bisa di-copy langsung

dialog_demo.py — demo semua dialog bawaan Live Demo 
import sys
from PySide6.QtWidgets import (
    QApplication, QWidget, QVBoxLayout, QPushButton,
    QLabel, QMessageBox, QFileDialog, QInputDialog
)

class DialogDemo(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Dialog Demo")
        self.setGeometry(100, 100, 350, 450)
        layout = QVBoxLayout(self)

        self.result = QLabel("Hasil dialog muncul di sini")
        self.result.setStyleSheet("background:#f0f0f0; padding:10px; border-radius:5px;")
        self.result.setWordWrap(True)
        layout.addWidget(self.result)

        for label, fn in [
            ("Information", self.show_info),
            ("Warning", self.show_warning),
            ("Critical", self.show_critical),
            ("Question (Yes/No)", self.show_question),
            ("Open File", self.open_file),
            ("Save File", self.save_file),
            ("Input Text", self.input_text),
            ("Input Integer", self.input_int),
            ("Select Item", self.select_item),
        ]:
            btn = QPushButton(label)
            btn.clicked.connect(fn)
            layout.addWidget(btn)

    def show_info(self):
        QMessageBox.information(self, "Info", "Operasi berhasil!")
        self.result.setText("Information dialog ditutup")

    def show_warning(self):
        QMessageBox.warning(self, "Peringatan", "Hati-hati dengan aksi ini!")
        self.result.setText("Warning dialog ditutup")

    def show_critical(self):
        QMessageBox.critical(self, "Error", "Terjadi kesalahan kritis!")
        self.result.setText("Critical dialog ditutup")

    def show_question(self):
        reply = QMessageBox.question(
            self, "Konfirmasi", "Apakah Anda yakin?",
            QMessageBox.Yes | QMessageBox.No | QMessageBox.Cancel,
            QMessageBox.No
        )
        if reply == QMessageBox.Yes:
            self.result.setText("Anda pilih: YES")
        elif reply == QMessageBox.No:
            self.result.setText("Anda pilih: NO")
        else:
            self.result.setText("Anda pilih: CANCEL")

    def open_file(self):
        path, _ = QFileDialog.getOpenFileName(
            self, "Buka File", "",
            "Text Files (*.txt);;Python Files (*.py);;All Files (*)"
        )
        self.result.setText(f"File: {path}" if path else "Dibatalkan")

    def save_file(self):
        path, _ = QFileDialog.getSaveFileName(
            self, "Simpan File", "dokumen.txt",
            "Text Files (*.txt);;All Files (*)"
        )
        self.result.setText(f"Simpan ke: {path}" if path else "Dibatalkan")

    def input_text(self):
        text, ok = QInputDialog.getText(self, "Input", "Masukkan nama:")
        self.result.setText(f"Nama: {text}" if ok and text else "Dibatalkan")

    def input_int(self):
        val, ok = QInputDialog.getInt(self, "Input", "Masukkan umur:", 20, 1, 120)
        self.result.setText(f"Umur: {val}" if ok else "Dibatalkan")

    def select_item(self):
        items = ["Python", "Java", "C++", "JavaScript", "Go"]
        item, ok = QInputDialog.getItem(self, "Pilih", "Bahasa favorit:", items, 0, False)
        self.result.setText(f"Pilihan: {item}" if ok else "Dibatalkan")

app = QApplication(sys.argv)
w = DialogDemo()
w.show()
sys.exit(app.exec())

        form = QFormLayout()

        self.nim.setPlaceholderText("Contoh: 12345678")

        self.jurusan = QComboBox()
        self.jurusan.addItems(["Teknik Informatika", "Sistem Informasi", "Teknik Komputer"])
        self.semester = QSpinBox()

        form.addRow("Jurusan:", self.jurusan)
        form.addRow("Semester:", self.semester)

        buttons.accepted.connect(self.validate)
        buttons.rejected.connect(self.reject)
        layout.addWidget(buttons)

    def validate(self):
        if not self.nim.text().strip():
            self.nim.setStyleSheet("border: 2px solid red;")
            self.nim.setFocus()
            return
        if not self.nama.text().strip():
            self.nama.setStyleSheet("border: 2px solid red;")
            self.nama.setFocus()
            return
        self.accept()

    def get_data(self):
        return {
            'nim': self.nim.text().strip(),
            'nama': self.nama.text().strip(),
            'jurusan': self.jurusan.currentText(),
            'semester': self.semester.value()
        }

import sys, os
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QTextEdit,
    QFileDialog, QMessageBox, QStyle
)
from PySide6.QtGui import QAction, QKeySequence

class Editor(QMainWindow):
    def __init__(self):
        super().__init__()
        self.current_file = None
        self.is_modified = False

        self.text = QTextEdit()
        self.setCentralWidget(self.text)
        self.text.textChanged.connect(self.on_changed)

        self.setup_actions()
        self.setup_menu()
        self.setup_toolbar()
        self.update_title()
        self.statusBar().showMessage("Ready")

    def setup_actions(self):
        s = self.style()
        self.act_new = QAction("&New", self)
        self.act_new.setIcon(s.standardIcon(QStyle.SP_FileIcon))
        self.act_new.setShortcut(QKeySequence.New)
        self.act_new.triggered.connect(self.new_file)

        self.act_open = QAction("&Open", self)
        self.act_open.setIcon(s.standardIcon(QStyle.SP_DialogOpenButton))
        self.act_open.setShortcut(QKeySequence.Open)
        self.act_open.triggered.connect(self.open_file)

        self.act_save = QAction("&Save", self)
        self.act_save.setIcon(s.standardIcon(QStyle.SP_DialogSaveButton))
        self.act_save.setShortcut(QKeySequence.Save)
        self.act_save.triggered.connect(self.save_file)

        self.act_saveas = QAction("Save &As", self)
        self.act_saveas.setShortcut(QKeySequence.SaveAs)
        self.act_saveas.triggered.connect(self.save_as)

    def setup_menu(self):
        m = self.menuBar().addMenu("&File")
        m.addAction(self.act_new)
        m.addAction(self.act_open)
        m.addSeparator()
        m.addAction(self.act_save)
        m.addAction(self.act_saveas)

    def setup_toolbar(self):
        tb = self.addToolBar("File")
        tb.addAction(self.act_new)
        tb.addAction(self.act_open)
        tb.addAction(self.act_save)

    def new_file(self):
        if self.maybe_save():
            self.text.clear()
            self.current_file = None
            self.is_modified = False
            self.update_title()

    def open_file(self):
        if self.maybe_save():
            path, _ = QFileDialog.getOpenFileName(
                self, "Buka File", "",
                "Text Files (*.txt);;Python Files (*.py);;All Files (*)"
            )
            if path:
                try:
                    with open(path, 'r', encoding='utf-8') as f:
                        self.text.setPlainText(f.read())
                    self.current_file = path
                    self.is_modified = False
                    self.update_title()
                    self.statusBar().showMessage(f"Dibuka: {path}", 3000)
                except Exception as e:
                    QMessageBox.critical(self, "Error", str(e))

    def save_file(self):
        if self.current_file:
            self._write(self.current_file)
        else:
            self.save_as()

    def save_as(self):
        path, _ = QFileDialog.getSaveFileName(
            self, "Simpan File",
            self.current_file or "untitled.txt",
            "Text Files (*.txt);;All Files (*)"
        )
        if path:
            self._write(path)

    def _write(self, path):
        try:
            with open(path, 'w', encoding='utf-8') as f:
                f.write(self.text.toPlainText())
            self.current_file = path
            self.is_modified = False
            self.update_title()
            self.statusBar().showMessage(f"Disimpan: {path}", 3000)
        except Exception as e:
            QMessageBox.critical(self, "Error", str(e))

    def maybe_save(self):
        if not self.is_modified:
            return True
        reply = QMessageBox.question(
            self, "Perubahan Belum Disimpan",
            "Simpan perubahan sebelum melanjutkan?",
            QMessageBox.Save | QMessageBox.Discard | QMessageBox.Cancel,
            QMessageBox.Save
        )
        if reply == QMessageBox.Save:
            self.save_file()
            return not self.is_modified
        return reply == QMessageBox.Discard

    def on_changed(self):
        self.is_modified = True
        self.update_title()

    def update_title(self):
        name = os.path.basename(self.current_file) if self.current_file else "Untitled"
        mark = " *" if self.is_modified else ""
        self.setWindowTitle(f"{name}{mark} — Text Editor")

    def closeEvent(self, event):
        event.accept() if self.maybe_save() else event.ignore()

app = QApplication(sys.argv)
w = Editor()
w.show()
sys.exit(app.exec())

def load_stylesheet(path):

import sys
from PySide6.QtWidgets import (
    QApplication, QMainWindow, QWidget, QVBoxLayout,
    QPushButton, QLabel, QTextEdit
)
from PySide6.QtGui import QAction

LIGHT = """
    QMainWindow, QWidget { background-color: #f5f6fa; color: #2c3e50; }
    QTextEdit { background: white; color: #2c3e50; border: 1px solid #ddd; border-radius: 6px; padding: 8px; }
    QPushButton { background-color: #3498db; color: white; border: none; border-radius: 6px; padding: 8px 16px; font-weight: bold; }
    QPushButton:hover { background-color: #2980b9; }
    QLabel { color: #2c3e50; font-size: 14px; }
    QMenuBar { background: #2c3e50; color: white; }
    QMenuBar::item:selected { background: #3498db; }
    QStatusBar { background: #ecf0f1; color: #7f8c8d; }
"""

DARK = """
    QMainWindow, QWidget { background-color: #1a1a2e; color: #e0e0e0; }
    QTextEdit { background: #16213e; color: #e0e0e0; border: 1px solid #0f3460; border-radius: 6px; padding: 8px; }
    QPushButton { background-color: #0f3460; color: #e0e0e0; border: 1px solid #e94560; border-radius: 6px; padding: 8px 16px; font-weight: bold; }
    QPushButton:hover { background-color: #e94560; color: white; }
    QLabel { color: #e0e0e0; font-size: 14px; }
    QMenuBar { background: #16213e; color: #e0e0e0; }
    QMenuBar::item:selected { background: #0f3460; }
    QStatusBar { background: #16213e; color: #7f8c8d; }
"""

class ThemeApp(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Light/Dark Theme Demo")
        self.setGeometry(100, 100, 600, 450)
        self.is_dark = False

        central = QWidget()
        self.setCentralWidget(central)
        layout = QVBoxLayout(central)

        self.theme_label = QLabel("Tema Aktif: Light Mode ☀️")
        self.theme_label.setStyleSheet("font-size: 16px; font-weight: bold; padding: 10px;")
        self.text = QTextEdit()
        self.text.setPlaceholderText("Ketik sesuatu...")
        self.toggle_btn = QPushButton("🌙 Switch to Dark Mode")
        self.toggle_btn.clicked.connect(self.toggle)

        layout.addWidget(self.theme_label)
        layout.addWidget(self.text)
        layout.addWidget(self.toggle_btn)

        # Shortcut keyboard untuk toggle
        act = QAction("Toggle Theme", self)
        act.setShortcut("Ctrl+Shift+T")
        act.triggered.connect(self.toggle)
        self.menuBar().addMenu("&View").addAction(act)

        self.statusBar().showMessage("Light Mode aktif")
        self.apply()

    def toggle(self):
        self.is_dark = not self.is_dark
        self.apply()

    def apply(self):
        if self.is_dark:
            QApplication.instance().setStyleSheet(DARK)
            self.theme_label.setText("Tema Aktif: Dark Mode 🌙")
            self.toggle_btn.setText("☀️ Switch to Light Mode")
            self.statusBar().showMessage("Dark Mode aktif")
        else:
            QApplication.instance().setStyleSheet(LIGHT)
            self.theme_label.setText("Tema Aktif: Light Mode ☀️")
            self.toggle_btn.setText("🌙 Switch to Dark Mode")
            self.statusBar().showMessage("Light Mode aktif")

app = QApplication(sys.argv)
w = ThemeApp()
w.show()
sys.exit(app.exec())
